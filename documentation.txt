create or replace function match_chunks(
  query_embedding vector(384),
  match_count int
)
returns table (
  id uuid,
  chunk_text text,
  title text,
  url text,
  similarity float
)
language sql stable
as $$
  select
    c.id,
    c.chunk_text,
    d.title,
    d.url,
    1 - (c.embedding <=> query_embedding) as similarity
  from chunks c
  join documents d on d.id = c.document_id
  order by c.embedding <=> query_embedding
  limit match_count;
$$;
